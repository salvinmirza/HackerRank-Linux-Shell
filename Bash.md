<h6>Let's Echo</h6>

~~~bash
  echo "HELLO"
~~~

<h6>Looping and Skipping</h6>

~~~bash
  for n in {1..99..2}
  do
      echo $n
  done
~~~

<h6>A Personalized Echo</h6>

~~~bash
    read name
    echo "Welcome $name"
~~~

<h6>Looping with Numbers</h6>

~~~bash
    for n in {1..50}
    do
        echo $n
    done
~~~

<h6>The World of Numbers</h6>

~~~bash
    read X
    read Y

    echo "$(($X + $Y))"
    echo "$(($X - $Y))"
    echo "$(($X * $Y))"
    echo "$(($X / $Y))"
~~~
